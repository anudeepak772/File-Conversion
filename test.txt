#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#define MAX 1000
// Stack implementation using array
typedef struct {
    int data[MAX];
    int top;
} Stack;
// Initialize the stack
void initStack(Stack *s) {
    s->top = -1;
}
// Check if stack is empty
int isEmpty(Stack *s) {
    return s->top == -1;
}
// Push to stack
void push(Stack *s, int value) {
    // Write your code here
    s->data[++s->top]=value;
}
// Pop from stack
int pop(Stack *s) {
    // Write your code here
    int value= s->data[s->top-1];
    free()
}
// Peek top of stack
int peek(Stack *s) {
    // Write your code here
    return s->data[s->top];
    
}
// Calculate total score from operations
int calculateScore(char operations[][10], int n) {
    // Write your code here
    Stack *s=(Stack *)malloc(sizeof(Stack*));
    initStack(s);
    int sum=0;
    for (int i=0;i<n;i++)
    {
        push(s,atoi(operations[i]));
    }
    while(!isEmpty(s))
    {
        sum=sum+pop(s);
    }
    return sum;
    
}
// Driver code
int main() {
    int n;
    scanf("%d", &n);
    char operations[n][10];
    for (int i = 0; i < n; i++) {
        scanf("%s", operations[i]);
    }
    int result = calculateScore(operations, n);
    printf("%d\n", result);
    return 0;
}
